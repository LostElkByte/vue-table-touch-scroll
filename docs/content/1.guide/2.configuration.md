## Site Configuration

```ts
export default defineAppConfig({
  shadcnDocs: {
    site: {
      name: 'My Documentation',
      description: 'A description of my documentation.',
      ogImage: '/cover.png',
    },
  },
})
```

## Theme Configuration

Customize the color theme and border radius:

```ts
export default defineAppConfig({
  shadcnDocs: {
    theme: {
      customizable: true,
      color: 'zinc',    // Color theme
      radius: 0.5,      // Border radius
    },
  },
})
```

Available colors: `zinc`, `slate`, `stone`, `gray`, `neutral`, `red`, `rose`, `orange`, `green`, `blue`, `yellow`, `violet`

## Header Configuration

Configure the navigation header:

```ts
export default defineAppConfig({
  shadcnDocs: {
    header: {
      title: 'My Docs',
      showTitle: true,
      darkModeToggle: true,
      logo: {
        light: '/logo.svg',
        dark: '/logo-dark.svg',
      },
      nav: [
        {
          title: 'Docs',
          links: [
            {
              title: 'Getting Started',
              to: '/guide/introduction',
              description: 'Learn the basics',
            },
          ],
        },
      ],
    },
  },
})
```

## Sidebar Configuration

Configure the aside navigation:

```ts
export default defineAppConfig({
  shadcnDocs: {
    aside: {
      useLevel: true,
      collapse: true,
      collapseLevel: 1,
    },
  },
})
```
