# Ant Design Vue 示例

Ant Design Vue 是 Ant Design 的 Vue 实现，提供了企业级的 UI 组件。

## 在线演示

::ComponentLoader{componentName="AntDesignVueExample" label="Ant Design Vue 示例"}
::

## 基础用法

```vue
<template>
  <div v-table-touch-scroll="{ uiLibrary: 'ant-design-vue' }">
    <a-table :columns="columns" :data-source="data" />
  </div>
</template>

<script setup lang="ts">
const columns = [
  {
    title: '姓名',
    dataIndex: 'name',
    key: 'name',
  },
  {
    title: '年龄',
    dataIndex: 'age',
    key: 'age',
  },
  {
    title: '地址',
    dataIndex: 'address',
    key: 'address',
  },
]

const data = [
  {
    key: '1',
    name: '张三',
    age: 32,
    address: '上海市普陀区金沙江路 1518 弄',
  },
  {
    key: '2',
    name: '李四',
    age: 42,
    address: '上海市普陀区金沙江路 1517 弄',
  },
]
</script>
```

## 固定列表格

```vue
<template>
  <div v-table-touch-scroll="{ uiLibrary: 'ant-design-vue' }">
    <a-table
      :columns="columns"
      :data-source="data"
      :scroll="{ x: 1500 }"
    />
  </div>
</template>

<script setup lang="ts">
const columns = [
  {
    title: 'Full Name',
    width: 100,
    dataIndex: 'name',
    key: 'name',
    fixed: 'left',
  },
  {
    title: 'Age',
    width: 100,
    dataIndex: 'age',
    key: 'age',
    fixed: 'left',
  },
  { title: 'Column 1', dataIndex: 'address', key: '1', width: 150 },
  { title: 'Column 2', dataIndex: 'address', key: '2', width: 150 },
  { title: 'Column 3', dataIndex: 'address', key: '3', width: 150 },
  { title: 'Column 4', dataIndex: 'address', key: '4', width: 150 },
  { title: 'Column 5', dataIndex: 'address', key: '5', width: 150 },
  {
    title: 'Action',
    key: 'operation',
    fixed: 'right',
    width: 100,
  },
]

const data = []
for (let i = 0; i < 100; i++) {
  data.push({
    key: i,
    name: `Edrward ${i}`,
    age: 32,
    address: `London Park no. ${i}`,
  })
}
</script>
```

## 完整示例代码

```vue
<script setup lang="ts">
import { useTableData } from './composables/useTableData'
import { useTableColumns } from './composables/useTableColumns'

const tableData = useTableData(50)
const columns = useTableColumns().map((col) => ({
  title: col.title,
  dataIndex: col.dataIndex,
  key: col.key,
  width: col.width,
  fixed: col.fixed,
}))
</script>

<template>
  <div
    v-table-touch-scroll="{ uiLibrary: 'ant-design-vue' }"
    class="border rounded-lg overflow-hidden"
  >
    <a-table
      :data-source="tableData"
      :columns="columns"
      :scroll="{ x: 1300, y: 400 }"
      size="small"
      bordered
    />
  </div>
</template>
```
