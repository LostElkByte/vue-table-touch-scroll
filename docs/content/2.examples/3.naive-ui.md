# Naive UI 示例

Naive UI 是一个 Vue 3 组件库，提供了丰富的主题定制能力。

## 在线演示

::ComponentLoader{componentName="NaiveUIExample" label="Naive UI 示例"}
::

## 基础用法

```vue
<template>
  <div v-table-touch-scroll="{ preset: 'naive-ui' }">
    <n-data-table :columns="columns" :data="data" />
  </div>
</template>

<script setup lang="ts">
import type { DataTableColumns } from 'naive-ui'

type RowData = {
  key: number
  name: string
  age: number
  address: string
}

const createColumns = (): DataTableColumns<RowData> => [
  {
    title: 'Name',
    key: 'name',
  },
  {
    title: 'Age',
    key: 'age',
  },
  {
    title: 'Address',
    key: 'address',
  },
]

const data: RowData[] = [
  {
    key: 0,
    name: 'John Brown',
    age: 32,
    address: 'New York No. 1 Lake Park',
  },
  {
    key: 1,
    name: 'Jim Green',
    age: 42,
    address: 'London No. 1 Lake Park',
  },
]

const columns = createColumns()
</script>
```

## 固定列表格

```vue
<template>
  <div v-table-touch-scroll="{ preset: 'naive-ui' }">
    <n-data-table
      :columns="columns"
      :data="data"
      :scroll-x="1800"
      :single-line="false"
    />
  </div>
</template>

<script setup lang="ts">
import type { DataTableColumns } from 'naive-ui'

const columns: DataTableColumns = [
  {
    title: 'Name',
    key: 'name',
    width: 200,
    fixed: 'left',
  },
  {
    title: 'Age',
    key: 'age',
    width: 100,
    fixed: 'left',
  },
  {
    title: 'Row1',
    key: 'row1',
    width: 200,
  },
  {
    title: 'Row2',
    key: 'row2',
    width: 200,
  },
  {
    title: 'Row3',
    key: 'row3',
    width: 200,
  },
  {
    title: 'Row4',
    key: 'row4',
    width: 200,
  },
  {
    title: 'Action',
    key: 'actions',
    width: 150,
    fixed: 'right',
  },
]

const data = Array.from({ length: 46 }).map((_, index) => ({
  key: index,
  name: `Name ${index}`,
  age: 20 + (index % 10),
  row1: `Row1 ${index}`,
  row2: `Row2 ${index}`,
  row3: `Row3 ${index}`,
  row4: `Row4 ${index}`,
}))
</script>
```

## 完整示例代码

```vue
<script setup lang="ts">
import type { DataTableColumns } from 'naive-ui'
import { useTableData, type TableRow } from './composables/useTableData'
import { useTableColumns } from './composables/useTableColumns'

const tableData = useTableData(50)
const rawColumns = useTableColumns()

const columns: DataTableColumns<TableRow> = rawColumns.map((col) => ({
  title: col.title,
  key: col.key,
  width: col.width,
  fixed: col.fixed,
}))
</script>

<template>
  <div
    v-table-touch-scroll="{ uiLibrary: 'naive-ui' }"
    class="border rounded-lg overflow-hidden"
  >
    <n-data-table
      :columns="columns"
      :data="tableData"
      :scroll-x="1300"
      :max-height="400"
      size="small"
      bordered
    />
  </div>
</template>
```
