## 站点配置

```ts
export default defineAppConfig({
  shadcnDocs: {
    site: {
      name: '我的文档',
      description: '我的文档描述。',
      ogImage: '/cover.png',
    },
  },
})
```

## 主题配置

自定义颜色主题和边框圆角：

```ts
export default defineAppConfig({
  shadcnDocs: {
    theme: {
      customizable: true,
      color: 'zinc',    // 颜色主题
      radius: 0.5,      // 边框圆角
    },
  },
})
```

可用颜色：`zinc`、`slate`、`stone`、`gray`、`neutral`、`red`、`rose`、`orange`、`green`、`blue`、`yellow`、`violet`

## 头部配置

配置导航头部：

```ts
export default defineAppConfig({
  shadcnDocs: {
    header: {
      title: '我的文档',
      showTitle: true,
      darkModeToggle: true,
      logo: {
        light: '/logo.svg',
        dark: '/logo-dark.svg',
      },
    },
  },
})
```
